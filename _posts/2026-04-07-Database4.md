# 📖오늘 배운 내용 

---

> **Python으로 2개의 데이터베이스 테이블(기본키와 외래키로 연결)을 연동하여, 도서 목록을 보여주고 특정 도서의 대출 기록을 조회하는 프로그램을 실습해보았습니다.**

**1. 데이터베이스 생성 (테이블 생성 및 연결)**

```sql
-- 데이터베이스 생성 및 선택
CREATE DATABASE library_db DEFAULT CHARACTER SET utf8mb4;
USE library_db;

-- 1. 부모 테이블 (books) 생성
-- id 컬럼을 기본키(PRIMARY KEY)로 설정하여 각 책의 고유 번호로 사용한다.
CREATE TABLE books (
    id INT AUTO_INCREMENT PRIMARY KEY,
    title VARCHAR(100) NOT NULL,
    author VARCHAR(50) NOT NULL
);

-- 2. 자식 테이블 (rentals) 생성
-- book_id 컬럼을 외래키(FOREIGN KEY)로 설정하여 books 테이블의 id를 참조하도록 연결한다.
CREATE TABLE rentals (
    rental_id INT AUTO_INCREMENT PRIMARY KEY,
    book_id INT,
    member VARCHAR(50) NOT NULL,
    rdate DATE NOT NULL,
    FOREIGN KEY (book_id) REFERENCES books(id) ON DELETE CASCADE
);
```

**2. 실습용 데이터 삽입**

```sql
-- 도서 데이터 삽입
INSERT INTO books (title, author) VALUES 
('파이썬 완전 정복', '김파이'), 
('데이터베이스 첫걸음', '이디비'),
('인공지능의 미래', '박에이');

-- 대출 기록 데이터 삽입 (1번 책 2번 대출, 2번 책 1번 대출)
INSERT INTO rentals (book_id, member, rdate) VALUES 
(1, '홍길동', '2026-04-01'),
(1, '김철수', '2026-04-05'),
(2, '이영희', '2026-04-07');
```

**3. Python 도서 대출 조회 프로그램 실습**

```python
import mysql.connector

# ① DB 연결
conn = mysql.connector.connect(
    host="localhost",
    user="root",
    password="본인비밀번호",   # 각자의 비밀번호로 변경
    database="library_db", 
    charset="utf8mb4"
)
cursor = conn.cursor()

# ② 도서 목록 먼저 출력 (어떤 책이 있는지 안내)
print("=== 도서 대출 조회 프로그램 ===")
print("\n[보유 도서 목록]")

cursor.execute("SELECT id, title, author FROM books")
for row in cursor.fetchall():
    print(f"  {row[0]}번. {row[1]} ({row[2]})")

# ③ 조회할 책 번호 입력받기
print()
book_id = input("조회할 책 번호를 입력하세요: ").strip()

# ④ 숫자 입력인지 확인 (유효성 검사)
if not book_id.isdigit():
    print("숫자로 입력해야 합니다.")
else:
    # ⑤ 해당 책의 대출 기록 조회 (두 테이블 연결)
    sql = """
    SELECT books.title, rentals.member, rentals.rdate
    FROM rentals, books
    WHERE rentals.book_id = books.id
      AND books.id = %s
    """
    # %s 자리에 book_id 값을 안전하게 넣는다
    cursor.execute(sql, (int(book_id),))
    rows = cursor.fetchall()

    # 조회 결과 출력
    if rows:
        print(f"\n[대출 기록]")
        for row in rows:
            print(f"  책: {row[0]} | 대출자: {row[1]} | 날짜: {row[2]}")
    else:
        print("\n대출 기록이 없거나 존재하지 않는 책 번호입니다.")

# ⑥ 연결 종료
cursor.close()
conn.close()
print("\n프로그램 종료")
```
