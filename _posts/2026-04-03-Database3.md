# 📖오늘 배운 내용

> **오늘은 파이썬(Python)과 MySQL을 연동하여, 데이터를 직접 추가하고 조회하며 삭제까지 할 수 있는 성적 관리 프로그램을 실습해 보았습니다.**

---

 **1. MySQL에서 데이터베이스와 테이블 생성**
 
 ```sql
# 1. 데이터베이스 생성 및 사용 준비
CREATE DATABASE my_database;
USE my_database;

# 2. 학생 성적을 저장할 테이블 생성
CREATE TABLE my_table (
    id INT PRIMARY KEY,   -- 고유 번호 (기본키)
    name VARCHAR(20),     -- 학생 이름 (최대 20자)
    subject VARCHAR(20),  -- 과목명 (최대 20자)
    score INT,            -- 점수 (정수형)
    grade INT             -- 등급 (정수형)
);
```

**2. 성적 관리 프로그램 파이썬 전체 코드**

```python
# 1. 필요한 도구(라이브러리) 가져오기
import mysql.connector

# 2. 데이터베이스 연결 설정
# 이 부분은 파이썬과 MySQL 서버를 '전화기'처럼 연결해주는 설정입니다.
try:
    conn = mysql.connector.connect(
        host="localhost",      # 내 컴퓨터에 있는 DB에 접속
        user="root",           # MySQL 아이디 (기본값: root)
        password="my_password", # 본인이 설정한 비밀번호 입력
        database="my_database"    # 미리 만들어둔 데이터베이스 이름
    )
    # cursor는 DB에 명령을 전달하고 결과를 받아오는 '심부름꾼' 역할을 합니다.
    cursor = conn.cursor()
except mysql.connector.Error as err:
    print(f"❌ DB 연결 실패: {err}")
    exit() # 연결 실패 시 프로그램 종료

# 3. 프로그램 메인 루프 (무한 반복)
while True:
    # 사용자에게 보여줄 메뉴판
    print("\n" + "="*60)
    print(" [메뉴] add(추가) | list(목록) | summary(통계) | delete(삭제) | exit(종료)")
    print("="*60)
    
    # 입력받은 명령어에서 공백을 제거(.strip)하고 소문자로 변환(.lower)해서 저장
    command = input("명령 입력: ").strip().lower()

    # --- [기능 1] 데이터 추가 (INSERT) ---
    if command == "add":
        id_input = input("번호 입력: ").strip()
        name = input("이름 입력: ").strip()
        subject = input("과목 입력: ").strip()
        score_input = input("점수 입력: ").strip()
        grade_input = input("학년 입력: ").strip()

        # .isdigit()은 입력값이 '숫자'로만 되어 있는지 검사하는 안전장치입니다.
        if not (id_input.isdigit() and score_input.isdigit() and grade_input.isdigit()):
            print("❗ 오류: 번호, 점수, 학년은 숫자로만 입력해야 합니다.")
            continue # 잘못 입력하면 처음(메뉴)으로 돌아갑니다.

        # SQL 쿼리 작성 (%s는 나중에 값을 채워넣을 빈칸입니다)
        sql = "INSERT INTO scores (id, name, subject, score, grade) VALUES (%s, %s, %s, %s, %s)"
        values = (int(id_input), name, subject, int(score_input), int(grade_input))
        
        try:
            cursor.execute(sql, values) # 명령 실행
            conn.commit()               # DB에 실제 저장 확정 (매우 중요!)
            print("✅ 성공: 데이터가 저장되었습니다.")
        except mysql.connector.Error as err:
            print(f"❌ 실패: {err}")

    # --- [기능 2] 목록 보기 (SELECT) ---
    elif command == "list":
        # scores 테이블의 모든 데이터를 가져옵니다.
        cursor.execute("SELECT * FROM scores")
        rows = cursor.fetchall() # 찾아온 데이터를 모두 rows 변수에 담습니다.

        if not rows:
            print("📢 알림: 저장된 데이터가 없습니다.")
            continue

        # 결과 출력 (보기 좋게 정렬해서 출력)
        print(f"\n{'ID':<5} {'이름':<10} {'과목':<10} {'점수':<5} {'학년':<5}")
        print("-" * 45)
        for row in rows:
            # row[0]은 첫 번째 컬럼(id), row[1]은 두 번째(name)... 순서입니다.
            print(f"{row[0]:<5} {row[1]:<10} {row[2]:<10} {row[3]:<5} {row[4]:<5}")

    # --- [기능 3] 통계 요약 (Group 함수) ---
    elif command == "summary":
        # COUNT(개수), SUM(합계), AVG(평균), MAX(최대), MIN(최소)를 한꺼번에 구합니다.
        cursor.execute("SELECT COUNT(*), SUM(score), AVG(score), MAX(score), MIN(score) FROM scores")
        res = cursor.fetchone() # 결과가 한 줄이므로 fetchone()을 씁니다.
        
        if res[0] == 0:
            print("📢 알림: 통계를 낼 데이터가 없습니다.")
        else:
            print(f"\n📊 [전체 통계]")
            print(f"- 총 데이터: {res[0]}개")
            print(f"- 점수 합계: {res[1]}점")
            print(f"- 점수 평균: {res[2]:.2f}점") # :.2f는 소수점 둘째 자리까지 표시
            print(f"- 최고 점수: {res[3]}점 / 최저 점수: {res[4]}점")

    # --- [기능 4] 데이터 삭제 (DELETE) ---
    elif command == "delete":
        del_id = input("삭제할 번호(ID) 입력: ").strip()
        
        # 숫자인지 먼저 확인해서 에러를 방지합니다.
        if not del_id.isdigit():
            print("❗ 오류: 삭제할 번호는 숫자로 입력해야 합니다.")
            continue

        # %s를 써서 특정 ID를 가진 행만 삭제합니다.
        sql = "DELETE FROM scores WHERE id = %s"
        try:
            cursor.execute(sql, (int(del_id),)) # 튜플 형태이므로 콤마(,) 필수!
            conn.commit() # 삭제 내역 확정
            
            # cursor.rowcount는 방금 명령으로 실제 삭제된 줄의 개수를 알려줍니다.
            if cursor.rowcount > 0:
                print(f"🗑️  성공: {del_id}번 데이터가 삭제되었습니다.")
            else:
                print(f"⚠️  알림: {del_id}번 ID를 찾을 수 없습니다.")
        except mysql.connector.Error as err:
            print(f"❌ 실패: {err}")

    # --- [기능 5] 프로그램 종료 ---
    elif command == "exit":
        print("👋 프로그램을 종료합니다. 이용해 주셔서 감사합니다!")
        break # while 루프를 빠져나갑니다.

    else:
        print("❓ 알 수 없는 명령어입니다. 메뉴에 있는 단어를 입력해 주세요.")

# 4. 자원 반납 (프로그램 종료 시 연결을 끊어줍니다)
cursor.close()
conn.close()
```
